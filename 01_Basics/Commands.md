# Initialising the empty application of Node
- init is the utility to create the package.json file
- npm init  (initialize the node application from node package manager)
- npm init -y (use the defaults)

# To install any packageand save it as dependency in package.json
- npm install <package>
- npm install express
- npm i dotenv

# To run the commands
- npm run start

# To create a file
- touch index.js

# To install nodemon for dev dep
- npm i -D nodemon/npm install --save-dev nodemon

# To work on the dev like usng the nodemon
- npm run dev
